File: cholesky_lstsq.f95 program cholesky_lstsq use mat double precision :: m, test, temp_real, s, h_temp, sum_h integer :: M_DIM, N_DIM, MAT_SIZE double precision, dimension(:, :), allocatable :: A_s, A, H double precision, dimension(:), allocatable :: y, b_s, b, solution character(2) :: do_math call getarg(1, do_math) m = 0 ! Get matrix A and dimensions from command line call PMAT(A_s, M_DIM, N_DIM) call PB(b_s, M_DIM, N_DIM) MAT_SIZE = min(M_DIM, N_DIM) ! Allocate arrays allocate(H(M_DIM, N_DIM)) allocate(A(M_DIM, N_DIM)) allocate(y(MAT_SIZE)) allocate(b(M_DIM)) allocate(solution(MAT_SIZE)) b = matmul(transpose(A_s), b_s) A = matmul(transpose(A_s), A_s) if (do_math == "1") then ! Factorize matrix and print results H = 0 do k = 1, MAT_SIZE h_temp = sum(H(k, :)**2); H(k, k) = sqrt(A(k, k) - h_temp); do ii = k+1, MAT_SIZE sum_h = sum(H(ii, :)*H(k, :)); H(ii, k) = (A(ii, k) - sum_h)/H(k, k); end do end do ! Solve Hy = b y(1) = b(1)/H(1, 1); do ii = 2, MAT_SIZE s = 0; do j = 1, MAT_SIZE-1 s = s + y(j)*H(ii, j); end do y(ii) = (b(ii) - s)/H(ii, ii); end do ! Solve Ht*x = y solution(MAT_SIZE) = y(MAT_SIZE)/H(MAT_SIZE, MAT_SIZE); do ii = MAT_SIZE-1, 1, -1 s = 0; do j = ii+1, MAT_SIZE s = s + solution(j)*H(j, ii); end do solution(ii) = (y(ii) - s)/H(ii, ii); end do end if do ii=1, MAT_SIZE write(*, "(F25.20, F25.20)"), solution(ii) end do end program cholesky_lstsq !
